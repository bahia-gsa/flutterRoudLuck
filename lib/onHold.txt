class PlayersTable extends StatefulWidget {

  final List<Map<String, dynamic>> players;

  PlayersTable(this.players);

  @override
  State<PlayersTable> createState() => _MyTableState();
}

class _MyTableState extends State<PlayersTable> {
  @override
  Widget build(BuildContext context) {
    return DataTable(
      columns: const [
        DataColumn(
          label: Center(
            child: Text(
              'NAME',
              style: TextStyle(
                fontWeight: FontWeight.bold,
                fontSize: 20,
              ),
            ),
          ),
        ),
        DataColumn(
          label: Center(
            child: Text(
              'EMAIL',
              style: TextStyle(
                fontWeight: FontWeight.bold,
                fontSize: 20,
              ),
              textAlign: TextAlign.center,
            ),
          ),
          numeric: true,
        ),
      ],
      rows: widget.players.map((player) => DataRow(
        cells: [
          DataCell(Center(child: Text(player['playerName'] ?? 'Unknown'))),
          DataCell(Center(child: Text(player['playerEmail'] ?? 'Unknown'))),
        ],
      )).toList(),
    );
  }
}
